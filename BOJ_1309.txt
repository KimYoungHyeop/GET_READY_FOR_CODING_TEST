n = int(input())
d = [[0]*3 for _ in range(n+1)]
d[0][0] = 1

#d[i][0] = i번 줄에 배치하지 않음
#d[i][1] = i번 줄의 왼쪽에만 배치함
#d[i][2] = i번 줄의 오른쪽에만 배치함
for i in range(1, n+1):
    d[i][0] = d[i-1][0] + d[i-1][1] + d[i-1][2];
    d[i][1] = d[i-1][0] + d[i-1][2];
    d[i][2] = d[i-1][0] + d[i-1][1];
    for j in range(3):
        d[i][j] %= 9901
        
print(sum(d[n]) % 9901)
