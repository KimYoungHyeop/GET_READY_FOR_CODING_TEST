n = int(input())
a = list(map(int,input().split()))
d = [0]*n
v = [-1]*n

for i in range(n):
    d[i] = 1
    for j in range(i):
        if a[j] < a[i] and d[j]+1 > d[i]:
            d[i] = d[j]+1
            v[i] = j
ans = max(d)
p = [i for i,x in enumerate(d) if x == ans][0]
print(ans)

def calc(p):
    if p == -1:
        return
    calc(v[p])
    print(a[p],end=' ')
calc(p)
print()
